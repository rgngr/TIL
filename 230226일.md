```java
import java.util.*;

class Solution {
    boolean solution(String s) {
        String[] arr = s.split("");
        Stack st = new Stack();
        
        try {
            for (int i=0; i<arr.length; i++) {
                if (arr[i].equals("(")) {
                    st.push("(");
                } else if (arr[i].equals(")")){
                    st.pop();
                }
            }
            if (st.isEmpty()) {
                return true;
            } else {
                return false;
            }
        } catch (EmptyStackException e) {
            return false;
        }
    }
}
```
